\printnomenclature

\nomenclature{$H$}{a hypothesis, a conjuction of clauses, note $H \in \mathcal{H}$}
\nomenclature{$\mathcal{H}$}{a set of hypotheses, a search space}
\nomenclature{$T$}{a theory, a conjunction of logical sentences}
\nomenclature{$\vee T$}{a disjunction of logical sentences $\phi \in T$}
\nomenclature{$\square$}{falsity}
%\nomenclature{$A_{heads}$}{a set of head atoms of the clauses in $A$}
%\nomenclature{$A_{bodies}$}{a set of body atoms of the clauses in $A$}
\nomenclature{$B$}{a set of logical sentences called background knowledge}
\nomenclature{$e$}{an example, $e \in E$}
\nomenclature{$E$}{a set of examples}
\nomenclature{$E^+$}{a conjunction of positive examples}
\nomenclature{$E^-$}{a disjunction of negative examples}

%set theory
\nomenclature{$\powerset{S}$}{a powerset of a set $S$}

%symbols
\nomenclature{$\triangle$}{a set difference}

%languages
\nomenclature{$L$}{a formal first-order language}
\nomenclature{$\mathcal{L}$}{a signature of the language $L$}

%logic
\nomenclature{$Cn$}{a consequence operator $Cn:L \to L$}
\nomenclature{$Cl$}{a classical consequence operator}

%models
